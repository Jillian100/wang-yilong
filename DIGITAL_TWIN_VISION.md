#### 3. Memory Layer (記憶層) - ChromaDB 向量庫

**RAG (Retrieval-Augmented Generation) 系統**:

```python
from chromadb import Client
from sentence_transformers import SentenceTransformer

class WangYilongMemorySystem:
    def __init__(self):
        self.client = Client()
        self.collection = self.client.create_collection("wang_yilong_memories")
        self.embedder = SentenceTransformer('all-MiniLM-L6-v2')

    def add_memory(self, text, metadata):
        """添加記憶"""
        embedding = self.embedder.encode([text])[0]
        self.collection.add(
            embeddings=[embedding.tolist()],
            documents=[text],
            metadatas=[metadata],
            ids=[f"memory_{metadata['timestamp']}"]
        )

    def search_memories(self, query, top_k=5):
        """語義搜索記憶"""
        query_embedding = self.embedder.encode([query])[0]
        results = self.collection.query(
            query_embeddings=[query_embedding.tolist()],
            n_results=top_k
        )
        return results
```
